            document.getElementById("drawerDesc").textContent = currentProduct.desc;
            document.getElementById("drawerIngredients").textContent = currentProduct.ingredients;
            document.getElementById("drawerNutrition").textContent = currentProduct.nutrition;
            document.getElementById("drawerQty").textContent = currentQty;

            document.getElementById("drawerOverlay").classList.add("open");
            document.getElementById("productDrawer").classList.add("open");
        }

        function closeDrawer() {
            document.getElementById("drawerOverlay").classList.remove("open");
            document.getElementById("productDrawer").classList.remove("open");
        }

        function changeQty(delta) {
            currentQty = Math.max(1, currentQty + delta);
            document.getElementById("drawerQty").textContent = currentQty;
        }

        function addCurrentToCart() {
            if (!currentProduct) return;
            for (let i = 0; i < currentQty; i++) adicionar(currentProduct.id);
            showToast(`${currentQty} × ${currentProduct.nome} adicionado(s)!`);
            closeDrawer();
        }

        function adicionar(id) {
            const prod = PRODUTOS.find(p => p.id === id);
            const exist = carrinho.find(p => p.id === id);
            if (exist) exist.qtd++;
            else carrinho.push({ ...prod, qtd: 1 });
            salvar();
            showToast(`☕ ${prod.nome} adicionado!`);
        }

        function salvar() {
            localStorage.setItem("pb_cart", JSON.stringify(carrinho));
            renderCart();
        }

        function renderCart() {
            const body = document.getElementById("cartBody");
            const totalEl = document.getElementById("cartTotal");
            const badge = document.getElementById("cartBadge");

            let total = 0;
            const count = carrinho.reduce((sum, p) => sum + p.qtd, 0);

            if (!carrinho.length) {
                body.innerHTML = `<div style="text-align:center;padding:80px 20px;color:var(--muted);">Seu carrinho está vazio</div>`;
                totalEl.textContent = "R$ 0,00";
                badge.textContent = "0";
                return;
            }

            let html = carrinho.map(p => {
                const itemTotal = p.preco * p.qtd;
                total += itemTotal;
                return `
                    <div style="display:flex;gap:16px;background:white;padding:16px;border-radius:20px;margin-bottom:16px;">
                        <img src="${p.img}" style="width:68px;height:68px;object-fit:cover;border-radius:12px;" alt="${p.nome}">
                        <div style="flex:1;">
                            <div style="font-weight:600;">${p.nome}</div>
                            <div style="color:var(--gold);">R$ ${itemTotal.toFixed(2).replace(".",",")}</div>
                            <div style="margin-top:12px;display:flex;gap:12px;align-items:center;">
                                <button onclick="diminuir(${p.id})" style="width:32px;height:32px;border:1px solid #ddd;border-radius:50%;">−</button>
                                <span style="font-weight:600;">${p.qtd}</span>
                                <button onclick="aumentar(${p.id})" style="width:32px;height:32px;border:1px solid #ddd;border-radius:50%;">+</button>
                            </div>
                        </div>
                        <button onclick="remover(${p.id})" style="color:var(--red);background:none;border:none;font-size:1.5rem;">✕</button>
                    </div>
                `;
            }).join('');

            body.innerHTML = html;
            totalEl.textContent = `R$ ${total.toFixed(2).replace(".",",")}`;
            badge.textContent = count;
        }

        function diminuir(id) { 
            const item = carrinho.find(p => p.id === id);
            if (item) {
                item.qtd--;
                if (item.qtd <= 0) remover(id);
                else salvar();
            }
        }

        function aumentar(id) {
            const item = carrinho.find(p => p.id === id);
            if (item) { item.qtd++; salvar(); }
        }

        function remover(id) {
            carrinho = carrinho.filter(p => p.id !== id);
            salvar();
        }

        function toggleCart() {
            const sidebar = document.getElementById("cartSidebar");
            const overlay = document.getElementById("cartOverlay");
            sidebar.classList.toggle("open");
            overlay.classList.toggle("open");
        }

        function finalizarPedido() {
            if (!carrinho.length) return showToast("Carrinho vazio");
            const addr = document.getElementById("addressInput").value.trim() || "Retirada";
            let msg = `Olá! Pedido:%0A`;
            carrinho.forEach(item => msg += `- ${item.nome} ×${item.qtd}%0A`);
            const total = carrinho.reduce((s, p) => s + p.preco * p.qtd, 0);
            msg += `%0ATotal: R$ ${total.toFixed(2)}%0AEndereço: ${encodeURIComponent(addr)}`;
            window.open(`https://wa.me/5515999999999?text=${msg}`, "_blank");
        }

        function showToast(msg) {
            const t = document.getElementById("toast");
            t.textContent = msg;
            t.classList.add("show");
            setTimeout(() => t.classList.remove("show"), 2600);
        }

        // Filters
        function initFilters() {
            const cats = ["todos","cafe","gelado","doce"];
            const labels = {todos:"Todos", cafe:"Cafés", gelado:"Gelados", doce:"Doces"};
            let html = '';
            cats.forEach(cat => {
                html += `<button onclick="filterCategory('${cat}')" class="filter-btn" style="padding:12px 26px;border-radius:9999px;border:2px solid var(--border);background:white;font-weight:500;">${labels[cat]}</button>`;
            });
            document.getElementById("filters").innerHTML = html;
        }

        function filterCategory(cat) {
            filtroAtual = cat;
            const filtered = cat === "todos" ? PRODUTOS : PRODUTOS.filter(p => p.cat === cat);
            renderMenu(filtered);
        }

        // Search
        document.getElementById("search").addEventListener("input", (e) => {
            const term = e.target.value.toLowerCase();
            const filtered = PRODUTOS.filter(p => 
                p.nome.toLowerCase().includes(term) || 
                (p.ingredients && p.ingredients.toLowerCase().includes(term))
            );
            renderMenu(filtered);
        });

        // Init
        window.onload = () => {
            renderFeatured();
            renderMenu();
            initFilters();
            renderCart();
        };
    </script>
