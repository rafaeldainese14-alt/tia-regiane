        }

            if (!currentProduct) return;

            for (let i = 0; i < currentQty; i++) {

                addToCart(currentProduct.id);

            }

            showToast(`${currentQty} × ${currentProduct.nome} adicionado(s)!`);

            closeDrawer();

        }

        function addToCart(id) {

            const prod = PRODUTOS.find(p => p.id === id);

            const existing = carrinho.find(item => item.id === id);

            

            if (existing) {

                existing.qtd = (existing.qtd || 1) + 1;

            } else {

                carrinho.push({ ...prod, qtd: 1 });

            }

            

            saveCart();

            showToast(`☕ ${prod.nome} adicionado`);

        }

        function saveCart() {

            localStorage.setItem("pb_cart", JSON.stringify(carrinho));

            renderCart();

        }

        function renderCart() {

            const body = document.getElementById("cartBody");

            const totalEl = document.getElementById("cartTotal");

            const badge = document.getElementById("cartBadge");

            let total = 0;

            let count = 0;

            if (carrinho.length === 0) {

                body.innerHTML = `<div style="text-align:center; padding:60px 20px; color:#9c8a75;">Seu carrinho está vazio</div>`;

                totalEl.textContent = "R$ 0,00";

                badge.textContent = "0";

                return;

            }

            let html = '';

            carrinho.forEach(item => {

                const itemTotal = item.preco * item.qtd;

                total += itemTotal;

                count += item.qtd;

                

                html += `

                    <div style="display:flex; gap:16px; background:white; padding:16px; border-radius:20px; margin-bottom:16px;">

                        <img src="${item.img}" style="width:70px; height:70px; object-fit:cover; border-radius:12px;">

                        <div style="flex:1;">

                            <div style="font-weight:600;">${item.nome}</div>

                            <div style="color:var(--gold); font-size:1.05rem;">R$ ${itemTotal.toFixed(2).replace('.', ',')}</div>

                            <div style="margin-top:12px; display:flex; align-items:center; gap:12px;">

                                <button onclick="updateCartQty(${item.id}, -1)" style="width:32px;height:32px;border:1px solid #ddd;border-radius:50%;">−</button>

                                <span>${item.qtd}</span>

                                <button onclick="updateCartQty(${item.id}, 1)" style="width:32px;height:32px;border:1px solid #ddd;border-radius:50%;">+</button>

                            </div>

                        </div>

                        <button onclick="removeFromCart(${item.id})" style="color:#e85d5d; background:none; border:none; font-size:1.4rem;">✕</button>

                    </div>

                `;

            });

            body.innerHTML = html;

            totalEl.textContent = `R$ ${total.toFixed(2).replace('.', ',')}`;

            badge.textContent = count;

        }

        function updateCartQty(id, delta) {

            const item = carrinho.find(i => i.id === id);

            if (item) {

                item.qtd += delta;

                if (item.qtd < 1) removeFromCart(id);

                else saveCart();

            }

        }

        function removeFromCart(id) {

            carrinho = carrinho.filter(i => i.id !== id);

            saveCart();

        }

        function toggleCart() {

            const sidebar = document.getElementById("cartSidebar");

            const overlay = document.getElementById("cartOverlay");

            sidebar.classList.toggle("open");

            overlay.classList.toggle("open");

        }

        function finalizarPedido() {

            if (!carrinho.length) {

                showToast("Carrinho vazio");

                return;

            }

            const addr = document.getElementById("addressInput").value.trim() || "Retirada na loja";

            const total = carrinho.reduce((sum, i) => sum + i.preco * i.qtd, 0);

            

            let msg = `Olá! Gostaria de pedir:%0A`;

            carrinho.forEach(item => {

                msg += `- ${item.nome} ×${item.qtd}%0A`;

            });

            msg += `%0ATotal: R$ ${total.toFixed(2)}%0AEndereço: ${encodeURIComponent(addr)}`;

            

            window.open(`https://wa.me/5515999999999?text=${msg}`, "_blank");

        }

        function showToast(message) {

            const toast = document.getElementById("toast");

            toast.textContent = message;

            toast.classList.add("show");

            setTimeout(() => toast.classList.remove("show"), 2800);

        }

        // Initialize filters

        function initFilters() {

            const cats = ["todos", "cafe", "gelado", "doce"];

            const labels = {todos: "Todos", cafe: "Cafés", gelado: "Gelados", doce: "Doces"};

            

            let html = '';

            cats.forEach(cat => {

                html += `<button onclick="filterCategory('${cat}')" class="filter-btn ${cat==='todos'?'active':''}" style="padding:10px 28px; border-radius:9999px; border:2px solid #e8d9c0; background:white;">${labels[cat]}</button>`;

            });

            document.getElementById("filters").innerHTML = html;

        }

        let activeFilter = "todos";

        function filterCategory(cat) {

            activeFilter = cat;

            document.querySelectorAll(".filter-btn").forEach(btn => {

                btn.classList.toggle("active", btn.textContent === (cat === "todos" ? "Todos" : (cat === "cafe" ? "Cafés" : cat === "gelado" ? "Gelados" : "Doces")));

            });

            

            const filtered = cat === "todos" ? PRODUTOS : PRODUTOS.filter(p => p.cat === cat);

            renderMenu(filtered);

        }

        // Search

        document.getElementById("searchInput").addEventListener("input", (e) => {

            const term = e.target.value.toLowerCase();

            const filtered = PRODUTOS.filter(p => 

                p.nome.toLowerCase().includes(term) || 

                (p.ingredients && p.ingredients.toLowerCase().includes(term))

            );

            renderMenu(filtered);

        });

        // Init

        window.onload = function() {

            renderFeatured();

            renderMenu();

            initFilters();

            renderCart();

        };
