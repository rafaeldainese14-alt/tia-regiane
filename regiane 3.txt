function toggleCart() {
  const sidebar = document.getElementById("cartSidebar");
  const overlay = document.getElementById("cartOverlay");

// ─── ADICIONAR ───
function adicionar(id) {
  const prod = PRODUTOS.find(p => p.id === id);
  const exist = carrinho.find(p => p.id === id);
  if (exist) exist.qtd++;
  else carrinho.push({ ...prod, qtd: 1 });
  salvar();
  showToast(`☕ ${prod.nome} adicionado!`);
}

// ─── DIMINUIR / AUMENTAR / REMOVER ───
function diminuir(id) {
  const item = carrinho.find(p => p.id === id);
  if (!item) return;
  item.qtd--;
  if (item.qtd <= 0) carrinho = carrinho.filter(p => p.id !== id);
  salvar();
}
function aumentar(id) {
  const item = carrinho.find(p => p.id === id);
  if (item) { item.qtd++; salvar(); }
}
function remover(id) {
  carrinho = carrinho.filter(p => p.id !== id);
  salvar();
}

// ─── SAVE & RENDER CART ───
function salvar() {
  localStorage.setItem("pb_cart", JSON.stringify(carrinho));
  renderCart();
}

function renderCart() {
  const body = document.getElementById("cartBody");
  const totalEl = document.getElementById("cartTotal");
  const badge = document.getElementById("cartBadge");

  const total = carrinho.reduce((s, p) => s + p.preco * p.qtd, 0);
  const qtdTotal = carrinho.reduce((s, p) => s + p.qtd, 0);

  badge.textContent = qtdTotal;
  if (totalEl) totalEl.textContent = `R$ ${total.toFixed(2).replace(".",",")}`;

  if (!carrinho.length) {
    body.innerHTML = `
      <div class="cart-empty">
        <div class="cart-empty-icon">🛒</div>
        <h3>Carrinho vazio</h3>
        <p>Adicione itens do cardápio para começar seu pedido.</p>
      </div>`;
    return;
  }

  body.innerHTML = carrinho.map(p => `
    <div class="cart-item">
      <img class="cart-item-img" src="${p.img}" alt="${p.nome}" loading="lazy">
      <div class="cart-item-info">
        <div class="cart-item-name">${p.nome}</div>
        <div class="cart-item-price">R$ ${(p.preco * p.qtd).toFixed(2).replace(".",",")}</div>
      </div>
      <div class="qty-controls">
        <button class="qty-btn" onclick="diminuir(${p.id})">−</button>
        <span class="qty-num">${p.qtd}</span>
        <button class="qty-btn" onclick="aumentar(${p.id})">+</button>
      </div>
      <button class="cart-remove" onclick="remover(${p.id})" aria-label="Remover">✕</button>
    </div>
  `).join("");
}

// ─── FINALIZAR ───
function finalizarPedido() {
  if (!carrinho.length) { showToast("⚠️ Seu carrinho está vazio"); return; }
  const addr = document.getElementById("addressInput").value.trim();
  if (!addr) {
    document.getElementById("addressInput").focus();
    showToast("⚠️ Informe seu endereço de entrega");
    return;
  }

  const itensTxt = carrinho.map(p => `${p.nome} x${p.qtd}`).join(", ");
  const total = carrinho.reduce((s, p) => s + p.preco * p.qtd, 0);
  const totalFmt = `R$ ${total.toFixed(2).replace(".",",")}`;

  const msg =
    `Olá Regiane,%0A%0AEu gostaria de encomendar: ${encodeURIComponent(itensTxt)}.%0A%0AMeu endereço é: ${encodeURIComponent(addr)}.%0A%0A💰 *Total: ${totalFmt}*`;

  window.open(`https://wa.me/5515999999999?text=${msg}`, "_blank");
}
